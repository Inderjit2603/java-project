16. Program to Find 3rd Largest Number in an array

Code


import java.util.Arrays;

public class ThirdLargestNumberExample {
    public static void main(String[] args) {
        int[] array = {12, 45, 78, 23, 56, 89, 34, 67};
                                        
        System.out.println("Original Array:");
        printArray(array);
                                        
        int thirdLargest = findThirdLargestNumber(array);
        System.out.println("\nThird Largest Number in the Array: " + thirdLargest);
    }
                                        
    private static int findThirdLargestNumber(int[] array) {
        if (array.length < 3) {
            System.out.println("Array does not have a 3rd largest number.");
            return Integer.MIN_VALUE; // Return a special value indicating no valid result
        }
                                        
        // Sort the array in descending order
        Arrays.sort(array);
                                        
        return array[array.length - 3];
    }
                                        
    private static void printArray(int[] array) {
        for (int value : array) {
            System.out.print(value + " ");
        }
    }
}

Output

Original Array: 

12 45 78 23 56 89 34 67  

Third Largest Number in the Array: 67
