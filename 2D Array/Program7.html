7. Program to display the upper triangular matrix

Code


import java.util.Scanner;

public class UpperTriangularMatrix {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
                                
        // Get the size of the square matrix
        System.out.print("Enter the size of the square matrix: ");
        int size = scanner.nextInt();
                                
        // Create a square matrix
        int[][] matrix = new int[size][size];
                                
        // Get elements for the matrix
        enterMatrixElements(scanner, matrix);
                                
        // Display the upper triangular matrix
        System.out.println("Original Matrix:");
        printMatrix(matrix);
                                
        System.out.println("Upper Triangular Matrix:");
        displayUpperTriangularMatrix(matrix);
                                
        scanner.close();
    }
                                
    // Function to enter matrix elements
    private static void enterMatrixElements(Scanner scanner, int[][] matrix) {
        System.out.println("Enter elements for the matrix:");
                                
        for (int i = 0; i < matrix.length; i++) {
            for (int j = 0; j < matrix[0].length; j++) {
                System.out.print("Enter element at position [" + i + "][" + j + "]: ");
                matrix[i][j] = scanner.nextInt();
            }
        }
    }
                                
    // Function to display the upper triangular matrix
    private static void displayUpperTriangularMatrix(int[][] matrix) {
        for (int i = 0; i < matrix.length; i++) {
            for (int j = 0; j < matrix[0].length; j++) {
                if (i <= j) {
                    System.out.print(matrix[i][j] + " ");
                } else {
                    System.out.print("0 ");
                }
            }
            System.out.println(); // Move to the next row
        }
    }
                                
    // Function to print a matrix
    private static void printMatrix(int[][] matrix) {
        for (int i = 0; i < matrix.length; i++) {
            for (int j = 0; j < matrix[0].length; j++) {
                System.out.print(matrix[i][j] + " ");
            }
            System.out.println(); // Move to the next row
        }
        System.out.println(); // Separate matrices by a blank line
    }
}
